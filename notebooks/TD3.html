def estOuvrante(c):
    return c in ('(', '[', '{')

def estFermante(c):
    return c in (')', ']', '}')

def fermante(c):
    fermantes = {
        '(': ')',
        '[': ']',
        '{': '}',
    }
    return fermantes[c]
    
def bonParenthesage(s):
    pile = []
    for c in s:
        if estOuvrante(c):
            pile.append(c)
        elif (estFermante(c)
                and (len(pile) == 0
                     or fermante(pile.pop()) != c)):
            return False
    return pile == []

def estOperateur(c):
    return c in ['+', '-', '/', '*']

def calcul(op, n, m):
    if op == '+':
        return n + m
    elif op == '-':
        return n - m
    elif op == '/':
        return n / m
    elif op == '*':
        return n * m
    else:
        raise ValueError('Mauvais op√©rateur')
        
def evaluation(s):
    acc_nombre = ''         # un accumulateur (pile) pour accumuler les chiffres d'un nombre
    acc_resultat = []       # l'accumulateur (pile) qui contient les operandes
    for c in s:
        if '0' <= c <= '9':
            acc_nombre += c
        elif len(acc_nombre) > 0:
            acc_resultat.append(int(acc_nombre))
            acc_nombre = ''
        if estOperateur(c):
            b = acc_resultat.pop()
            a = acc_resultat.pop()
            acc_resultat.append(calcul(c, a, b))
    res = acc_resultat.pop()
    assert len(acc_resultat) == 0, "Pile non vide"
    return res
